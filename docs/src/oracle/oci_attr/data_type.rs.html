39/// Attribute value used in [`DataType`]. You have no need to use this except implementing [`DataType`] for your type.
40pub struct AttrValue {
41    ctxt: Context,
42    handle: Handle,
43    attr_num: u32,
44}
45
46impl AttrValue {
47    pub(crate) fn from_conn(conn: &Connection, handle_type: u32, attr_num: u32) -> AttrValue {
48        AttrValue {
49            ctxt: conn.ctxt().clone(),
50            handle: Handle::Conn(conn.handle(), handle_type),
51            attr_num,
52        }
53    }
54
55    pub(crate) fn from_stmt(stmt: &Statement, attr_num: u32) -> AttrValue {
56        AttrValue {
57            ctxt: stmt.ctxt().clone(),
58            handle: Handle::Stmt(stmt.handle()),
59            attr_num,
60        }
61    }
62
63    fn ctxt(&self) -> &Context {
64        &self.ctxt
65    }
66
67    fn set_ptr_len(&mut self, ptr: *mut c_void, len: u32) -> Result<()> {
68        match &self.handle {
69            Handle::Conn(handle, handle_type) => {
70                chkerr!(
71                    self.ctxt(),
72                    dpiConn_setOciAttr(*handle, *handle_type, self.attr_num, ptr, len)
73                );
74            }
75            Handle::Stmt(handle) => {
76                chkerr!(
77                    self.ctxt(),
78                    dpiStmt_setOciAttr(*handle, self.attr_num, ptr, len)
79                );
80            }
81        }
82        Ok(())
83    }
84
85    /// Sets a value to the attribute.
86    ///
87    /// # Safety
88    ///
89    /// If incorrect type parameter `T` is specified,
90    /// its behavior is undefined. It may cause an access violation.
91    pub unsafe fn set<T>(&mut self, val: &T::Type) -> Result<()>
92    where
93        T: DataType,
94    {
95        <T>::set(self, val)
96    }
97
98    unsafe fn set_ub1(&mut self, val: u8) -> Result<()> {
99        self.set_ptr_len(&val as *const u8 as *mut c_void, 1)
100    }
101
102    unsafe fn set_ub2(&mut self, val: u16) -> Result<()> {
103        self.set_ptr_len(&val as *const u16 as *mut c_void, 2)
104    }
105
106    unsafe fn set_ub4(&mut self, val: u32) -> Result<()> {
107        self.set_ptr_len(&val as *const u32 as *mut c_void, 4)
108    }
109
110    unsafe fn set_ub8(&mut self, val: u64) -> Result<()> {
111        self.set_ptr_len(&val as *const u64 as *mut c_void, 8)
112    }
113
114    unsafe fn set_bool(&mut self, val: bool) -> Result<()> {
115        let val = i32::from(val);
116        self.set_ptr_len(&val as *const i32 as *mut c_void, 4)
117    }
118
119    unsafe fn set_pointer(&mut self, val: *mut c_void) -> Result<()> {
120        self.set_ptr_len(
121            &val as *const *mut c_void as *mut c_void,
122            mem::size_of::<*mut c_void>() as u32,
123        )
124    }
125
126    unsafe fn set_string(&mut self, val: &str) -> Result<()> {
127        let mut vec = Vec::with_capacity(val.len() + 1);
128        vec.extend_from_slice(val.as_bytes());
129        vec.push(0);
130        self.set_ptr_len(vec.as_ptr() as *mut c_void, val.len() as u32)
131    }
132
133    unsafe fn set_binary(&mut self, val: &[u8]) -> Result<()> {
134        self.set_ptr_len(val.as_ptr() as *mut c_void, val.len() as u32)
135    }
136
137    fn get_data_buffer(&self) -> Result<(dpiDataBuffer, u32)> {
138        let mut buf = unsafe { mem::zeroed() };
139        let mut len = 0;
140        match &self.handle {
141            Handle::Conn(handle, handle_type) => {
142                chkerr!(
143                    self.ctxt(),
144                    dpiConn_getOciAttr(*handle, *handle_type, self.attr_num, &mut buf, &mut len)
145                );
146            }
147            Handle::Stmt(handle) => {
148                chkerr!(
149                    self.ctxt(),
150                    dpiStmt_getOciAttr(*handle, self.attr_num, &mut buf, &mut len)
151                );
152            }
153        }
154        Ok((buf, len))
155    }
156
157    /// Gets a value from the attribute.
158    ///
159    /// # Safety
160    ///
161    /// If incorrect type parameter `T` is specified,
162    /// its behavior is undefined. It may cause an access violation.
163    pub unsafe fn get<T>(self) -> Result<<T::Type as ToOwned>::Owned>
164    where
165        T: DataType,
166    {
167        <T>::get(self)
168    }
169
170    unsafe fn get_ub1(&self) -> Result<u8> {
171        Ok(self.get_data_buffer()?.0.asUint8)
172    }
173
174    unsafe fn get_ub2(&self) -> Result<u16> {
175        Ok(self.get_data_buffer()?.0.asUint16)
176    }
177
178    unsafe fn get_ub4(&self) -> Result<u32> {
179        Ok(self.get_data_buffer()?.0.asUint32)
180    }
181
182    unsafe fn get_ub8(&self) -> Result<u64> {
183        Ok(self.get_data_buffer()?.0.asUint64)
184    }
185
186    unsafe fn get_bool(&self) -> Result<bool> {
187        Ok(self.get_data_buffer()?.0.asBoolean != 0)
188    }
189
190    unsafe fn get_pointer(&self) -> Result<*mut c_void> {
191        Ok(self.get_data_buffer()?.0.asRaw)
192    }
193
194    unsafe fn get_string(&self) -> Result<String> {
195        let val = self.get_data_buffer()?;
196        Ok(to_rust_str(val.0.asString, val.1))
197    }
198
199    unsafe fn get_binary(&self) -> Result<Vec<u8>> {
200        let val = self.get_data_buffer()?;
201        let s = slice::from_raw_parts(val.0.asRaw as *const u8, val.1.try_into()?);
202        Ok(s.to_vec())
203    }
204}

206/// A trait to get and set OCI attributes as rust types. You have no need to use this except implementing [`OciAttr`] for your type.
207#[allow(clippy::missing_safety_doc)]
208pub unsafe trait DataType {
209    type Type: ToOwned + ?Sized;
210
211    unsafe fn set(attr: &mut AttrValue, val: &Self::Type) -> Result<()>;
212    unsafe fn get(attr: AttrValue) -> Result<<Self::Type as ToOwned>::Owned>;
213}

311/// A type to get and set u64 microsecond attribute values as [`Duration`].
312///
313/// This is a data type for [`CallTime`](super::CallTime`).
314pub struct DurationUsecU64 {
315    _unused: [usize; 0],
316}
317
318unsafe impl DataType for DurationUsecU64 {
319    type Type = Duration;
320
321    unsafe fn set(_attr: &mut AttrValue, _val: &Duration) -> Result<()> {
322        // OCI_ATTR_CALL_TIME is read-only attribute.
323        // No need to implement `set`.
324        //
325        // Here is an example when `set` is required to be implemented.
326        //
327        //     let usecs = val.as_micros().try_into()?;
328        //     if usecs == 0 && !val.is_zero() {
329        //         return Err(Error::OutOfRange(format!("too short duration: {:?}", val)));
330        //     }
331        //     attr.set::<u64>(&usecs) // set microseconds as u64
332        unimplemented!()
333    }
334
335    unsafe fn get(attr: AttrValue) -> Result<Duration> {
336        let val = attr.get::<u64>()?;
337        Ok(Duration::from_micros(val))
338    }
339}

341/// A type corresponding to the `init.ora` parameter [`MAX_STRING_SIZE`]
342///
343/// This is a data type for [`CallTime`](super::VarTypeMaxLenCompat`).
344///
345/// [`MAX_STRING_SIZE`]: https://www.oracle.com/pls/topic/lookup?ctx=dblatest&id=GUID-D424D23B-0933-425F-BC69-9C0E6724693C
346#[derive(Debug, Clone)]
347pub enum MaxStringSize {
348    /// The maximum size is 4000 bytes for `VARCHAR2` and `NVARCHAR`, 2000 bytes for `RAW`.
349    Standard,
350    /// The maximum size is 32767 bytes for `VARCHAR2`, `NVARCHAR` and `RAW`.
351    Extended,
352}
353
354unsafe impl DataType for MaxStringSize {
355    type Type = MaxStringSize;
356
357    unsafe fn set(_attr: &mut AttrValue, _val: &MaxStringSize) -> Result<()> {
358        // OCI_ATTR_VARTYPE_MAXLEN_COMPAT is read-only attribute.
359        // No need to implement `set`.
360        unimplemented!()
361    }
362
363    unsafe fn get(attr: AttrValue) -> Result<MaxStringSize> {
364        // Constants defined in `oci.h` included in Oracle Instant Client SDK
365        const OCI_ATTR_MAXLEN_COMPAT_STANDARD: u8 = 1;
366        const OCI_ATTR_MAXLEN_COMPAT_EXTENDED: u8 = 2;
367
368        match attr.get::<u8>()? {
369            OCI_ATTR_MAXLEN_COMPAT_STANDARD => Ok(MaxStringSize::Standard),
370            OCI_ATTR_MAXLEN_COMPAT_EXTENDED => Ok(MaxStringSize::Extended),
371            x => Err(Error::internal_error(format!(
372                "invalid MaxStringSize {}",
373                x
374            ))),
375        }
376    }

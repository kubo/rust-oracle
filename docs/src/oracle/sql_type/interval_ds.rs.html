27/// Oracle-specific [Interval Day to Second][INTVL_DS] data type.
28///
29/// [INTVL_DS]: https://www.oracle.com/pls/topic/lookup?ctx=dblatest&id=GUID-FD8C41B7-8CDC-4D02-8E6B-5250416BC17D
30///
31/// This struct doesn't have arithmetic methods and they won't be added to avoid
32/// reinventing the wheel. If you need methods such as adding an interval to a
33/// timestamp, enable `chrono` feature and use [chrono::Duration][] instead.
34///
35/// [chrono::Duration]: https://docs.rs/chrono/0.4/chrono/struct.Duration.html
36///
37/// # Examples
38///
39/// ```
40/// # use oracle::*; use oracle::sql_type::*;
41///
42/// // Create an interval by new().
43/// let intvl1 = IntervalDS::new(1, 2, 3, 4, 500000000)?;
44///
45/// // All arguments must be zero or negative to create a negative interval.
46/// let intvl2 = IntervalDS::new(-1, -2, -3, -4, -500000000)?;
47///
48/// // Convert to string.
49/// assert_eq!(intvl1.to_string(), "+000000001 02:03:04.500000000");
50/// assert_eq!(intvl2.to_string(), "-000000001 02:03:04.500000000");
51///
52/// // Create an interval with leading field and fractional second precisions.
53/// let intvl3 = IntervalDS::new(1, 2, 3, 4, 500000000)?.and_prec(2, 3)?;
54///
55/// // The string representation depends on the precisions.
56/// assert_eq!(intvl3.to_string(), "+01 02:03:04.500");
57///
58/// // Precisions are ignored when intervals are compared.
59/// assert!(intvl1 == intvl3);
60///
61/// // Create an interval from string.
62/// let intvl4: IntervalDS = "+1 02:03:04.50".parse()?;
63///
64/// // The precisions are determined by number of decimal digits in the string.
65/// assert_eq!(intvl4.lfprec(), 1);
66/// assert_eq!(intvl4.fsprec(), 2);
67/// # Ok::<(), Error>(())
68/// ```
69///
70/// Fetch and bind interval values.
71///
72/// ```no_run
73/// # use oracle::*; use oracle::sql_type::*;
74/// let conn = Connection::connect("scott", "tiger", "")?;
75///
76/// // Fetch IntervalDS
77/// let sql = "select interval '+01 02:03:04.500' day to second(3) from dual";
78/// let intvl = conn.query_row_as::<IntervalDS>(sql, &[])?;
79/// assert_eq!(intvl.to_string(), "+01 02:03:04.500");
80///
81/// // Bind IntervalDS
82/// let sql = "begin \
83///              :outval := to_timestamp('2017-08-09', 'yyyy-mm-dd') + :inval; \
84///            end;";
85/// let mut stmt = conn.statement(sql).build()?;
86/// stmt.execute(&[&OracleType::Timestamp(3), // bind null as timestamp(3)
87///                &intvl, // bind the intvl variable
88///               ])?;
89/// let outval: Timestamp = stmt.bind_value(1)?; // get the first bind value.
90/// // 2017-08-09 + (1 day, 2 hours, 3 minutes and 4.5 seconds)
91/// assert_eq!(outval.to_string(), "2017-08-10 02:03:04.500");
92/// # Ok::<(), Error>(())
93/// ```
94#[derive(Debug, Clone, Copy)]
95pub struct IntervalDS {
96    pub(crate) intvl: dpiIntervalDS,
97    lfprec: u8,
98    fsprec: u8,
99}
100
101impl IntervalDS {
102    fn check(days: i32, hours: i32, minutes: i32, seconds: i32, nanoseconds: i32) -> Result<()> {
103        let errmsg = if !(-999999999..=999999999).contains(&days) {
104            "days must be between -999999999 and 999999999"
105        } else if !(-23..=23).contains(&hours) {
106            "hours must be between -23 and 23"
107        } else if !(-59..=59).contains(&minutes) {
108            "minutes must be between -59 and 59"
109        } else if !(-59..=59).contains(&seconds) {
110            "seconds must be between -59 and 59"
111        } else if !(-999999999..=999999999).contains(&nanoseconds) {
112            "nanoseconds must be between -999999999 and 999999999"
113        } else if days >= 0 && hours >= 0 && minutes >= 0 && seconds >= 0 && nanoseconds >= 0 {
114            // all members are zero or positive.
115            return Ok(());
116        } else if days <= 0 && hours <= 0 && minutes <= 0 && seconds <= 0 && nanoseconds <= 0 {
117            // all members are zero or negative.
118            return Ok(());
119        } else {
120            "days, hours, minutes, seconds and nanoseconds must be zeor or positive; or zero or negative"
121        };
122        Err(Error::out_of_range(format!(
123            "{errmsg} but days={days}, hours={hours}, minutes={minutes}, seconds={seconds}, nanoseconds={nanoseconds}"
124        )))
125    }
126
127    pub(crate) fn from_dpi_interval_ds(
128        it: &dpiIntervalDS,
129        oratype: &OracleType,
130    ) -> Result<IntervalDS> {
131        let (lfprec, fsprec) = match *oratype {
132            OracleType::IntervalDS(lfprec, fsprec) => (lfprec, fsprec),
133            _ => (0, 0),
134        };
135        IntervalDS::new(it.days, it.hours, it.minutes, it.seconds, it.fseconds)?
136            .and_prec(lfprec, fsprec)
137    }
138
139    /// Creates a new IntervalDS.
140    ///
141    /// Valid values are:
142    ///
143    /// | argument | valid values |
144    /// |---|---|
145    /// | `days` | -999999999 to 999999999 |
146    /// | `hours` | -23 to 23 |
147    /// | `minutes` | -59 to 59 |
148    /// | `seconds` | -59 to 59 |
149    /// | `nanoseconds` | -999999999 to 999999999 |
150    ///
151    /// All arguments must be zero or positive to create a positive interval.
152    /// All arguments must be zero or negative to create a negative interval.
153    pub fn new(
154        days: i32,
155        hours: i32,
156        minutes: i32,
157        seconds: i32,
158        nanoseconds: i32,
159    ) -> Result<IntervalDS> {
160        Self::check(days, hours, minutes, seconds, nanoseconds)?;
161        Ok(IntervalDS {
162            intvl: dpiIntervalDS {
163                days,
164                hours,
165                minutes,
166                seconds,
167                fseconds: nanoseconds,
168            },
169            lfprec: 9,
170            fsprec: 9,
171        })
172    }
173
174    /// Creates a new IntervalDS with precisions.
175    ///
176    /// `lfprec` and `fsprec` are leading field precision and fractional second
177    /// precision respectively.
178    /// The precisions affect text representation of IntervalDS.
179    /// They don't affect comparison.
180    pub fn and_prec(&self, lfprec: u8, fsprec: u8) -> Result<IntervalDS> {
181        if lfprec > 9 {
182            Err(Error::out_of_range(format!(
183                "lfprec must be 0 to 9 but {}",
184                lfprec
185            )))
186        } else if fsprec > 9 {
187            Err(Error::out_of_range(format!(
188                "fsprec must be 0 to 9 but {}",
189                fsprec
190            )))
191        } else {
192            Ok(IntervalDS {
193                lfprec,
194                fsprec,
195                ..*self
196            })
197        }
198    }
199
200    /// Returns days component.
201    pub fn days(&self) -> i32 {
202        self.intvl.days
203    }
204
205    /// Returns hours component.
206    pub fn hours(&self) -> i32 {
207        self.intvl.hours
208    }
209
210    /// Returns minutes component.
211    pub fn minutes(&self) -> i32 {
212        self.intvl.minutes
213    }
214
215    /// Returns seconds component.
216    pub fn seconds(&self) -> i32 {
217        self.intvl.seconds
218    }
219
220    /// Returns nanoseconds component.
221    pub fn nanoseconds(&self) -> i32 {
222        self.intvl.fseconds
223    }
224
225    /// Returns leading field precision.
226    pub fn lfprec(&self) -> u8 {
227        self.lfprec
228    }
229
230    /// Returns fractional second precision.
231    pub fn fsprec(&self) -> u8 {
232        self.fsprec
233    }
234}

236impl cmp::PartialEq for IntervalDS {
237    fn eq(&self, other: &Self) -> bool {
238        self.intvl.days == other.intvl.days
239            && self.intvl.hours == other.intvl.hours
240            && self.intvl.minutes == other.intvl.minutes
241            && self.intvl.seconds == other.intvl.seconds
242            && self.intvl.fseconds == other.intvl.fseconds
243    }
244}

246impl fmt::Display for IntervalDS {
247    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
248        if self.intvl.days < 0
249            || self.intvl.hours < 0
250            || self.intvl.minutes < 0
251            || self.intvl.seconds < 0
252            || self.intvl.fseconds < 0
253        {
254            write!(f, "-")?;
255        } else {
256            write!(f, "+")?;
257        };
258        let days = self.intvl.days.abs();
259        match self.lfprec {
260            2 => write!(f, "{:02}", days)?,
261            3 => write!(f, "{:03}", days)?,
262            4 => write!(f, "{:04}", days)?,
263            5 => write!(f, "{:05}", days)?,
264            6 => write!(f, "{:06}", days)?,
265            7 => write!(f, "{:07}", days)?,
266            8 => write!(f, "{:08}", days)?,
267            9 => write!(f, "{:09}", days)?,
268            _ => write!(f, "{}", days)?,
269        };
270        write!(
271            f,
272            " {:02}:{:02}:{:02}",
273            self.intvl.hours.abs(),
274            self.intvl.minutes.abs(),
275            self.intvl.seconds.abs()
276        )?;
277        let nsec = self.intvl.fseconds.abs();
278        match self.fsprec {
279            1 => write!(f, ".{:01}", nsec / 100000000),
280            2 => write!(f, ".{:02}", nsec / 10000000),
281            3 => write!(f, ".{:03}", nsec / 1000000),
282            4 => write!(f, ".{:04}", nsec / 100000),
283            5 => write!(f, ".{:05}", nsec / 10000),
284            6 => write!(f, ".{:06}", nsec / 1000),
285            7 => write!(f, ".{:07}", nsec / 100),
286            8 => write!(f, ".{:08}", nsec / 10),
287            9 => write!(f, ".{:09}", nsec),
288            _ => Ok(()),
289        }
290    }
291}

293impl str::FromStr for IntervalDS {
294    type Err = ParseOracleTypeError;
295
296    fn from_str(s: &str) -> result::Result<Self, Self::Err> {
297        let err = || ParseOracleTypeError::new("IntervalDS");
298        let mut s = Scanner::new(s);
299        let minus = match s.char() {
300            Some('+') => {
301                s.next();
302                false
303            }
304            Some('-') => {
305                s.next();
306                true
307            }
308            _ => false,
309        };
310        let days = s.read_digits().ok_or_else(err)? as i32;
311        let lfprec = s.ndigits();
312        if let Some(' ') = s.char() {
313            s.next();
314        } else {
315            return Err(err());
316        }
317        let hours = s.read_digits().ok_or_else(err)? as i32;
318        if let Some(':') = s.char() {
319            s.next();
320        } else {
321            return Err(err());
322        }
323        let minutes = s.read_digits().ok_or_else(err)? as i32;
324        if let Some(':') = s.char() {
325            s.next();
326        } else {
327            return Err(err());
328        }
329        let seconds = s.read_digits().ok_or_else(err)? as i32;
330        let mut nsecs = 0;
331        let mut fsprec = 0;
332        if let Some('.') = s.char() {
333            s.next();
334            nsecs = s.read_digits().ok_or_else(err)? as i32;
335            let ndigit = s.ndigits();
336            fsprec = ndigit;
337            match ndigit.cmp(&9) {
338                Ordering::Less => nsecs *= 10i32.pow(9 - ndigit),
339                Ordering::Equal => (),
340                Ordering::Greater => {
341                    nsecs /= 10i32.pow(ndigit - 9);
342                    fsprec = 9;
343                }
344            }
345        }
346        if s.char().is_some() {
347            return Err(err());
348        }
349        IntervalDS::new(
350            if minus { -days } else { days },
351            if minus { -hours } else { hours },
352            if minus { -minutes } else { minutes },
353            if minus { -seconds } else { seconds },
354            if minus { -nsecs } else { nsecs },
355        )
356        .map_err(|_| err())?
357        .and_prec(lfprec as u8, fsprec as u8)
358        .map_err(|_| err())
359    }
360}

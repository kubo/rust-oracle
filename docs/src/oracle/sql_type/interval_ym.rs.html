27/// Oracle-specific [Interval Year to Month][INTVL_YM] data type.
28///
29/// [INTVL_YM]: https://www.oracle.com/pls/topic/lookup?ctx=dblatest&id=GUID-517CEB46-C6FA-4B94-9299-5BBB5A58CF7B
30///
31/// # Examples
32///
33/// ```
34/// # use oracle::*; use oracle::sql_type::*;
35/// // Create an interval by new().
36/// let intvl1 = IntervalYM::new(2, 3)?;
37///
38/// // All arguments must be zero or negative to create a negative interval.
39/// let intvl2 = IntervalYM::new(-2, -3)?;
40///
41/// // Convert to string.
42/// assert_eq!(intvl1.to_string(), "+000000002-03");
43/// assert_eq!(intvl2.to_string(), "-000000002-03");
44///
45/// // Create an interval with precision.
46/// let intvl3 = IntervalYM::new(2, 3)?.and_prec(3)?;
47///
48/// // The string representation depends on the precisions.
49/// assert_eq!(intvl3.to_string(), "+002-03");
50///
51/// // Precisions are ignored when intervals are compared.
52/// assert!(intvl1 == intvl3);
53///
54/// // Create an interval from string.
55/// let intvl4: IntervalYM = "+002-3".parse()?;
56///
57/// // The precision is determined by number of decimal digits in the string.
58/// assert_eq!(intvl4.precision(), 3);
59/// # Ok::<(), Error>(())
60/// ```
61///
62/// Fetch and bind interval values.
63///
64/// ```no_run
65/// # use oracle::*; use oracle::sql_type::*;
66/// let conn = Connection::connect("scott", "tiger", "")?;
67///
68/// // Fetch IntervalYM
69/// let sql = "select interval '+02-03' year to month from dual";
70/// let intvl = conn.query_row_as::<IntervalYM>(sql, &[])?;
71/// assert_eq!(intvl.to_string(), "+02-03");
72///
73/// // Bind IntervalYM
74/// let sql = "begin \
75///              :outval := to_timestamp('2017-08-09', 'yyyy-mm-dd') + :inval; \
76///            end;";
77/// let mut stmt = conn.statement(sql).build()?;
78/// stmt.execute(&[&OracleType::Date, // bind null as date
79///                &intvl, // bind the intvl variable
80///               ])?;
81/// let outval: Timestamp = stmt.bind_value(1)?; // get the first bind value.
82/// // 2017-08-09 + (2 years and 3 months)
83/// assert_eq!(outval.to_string(), "2019-11-09 00:00:00");
84/// # Ok::<(), Error>(())
85/// ```
86#[derive(Debug, Clone, Copy)]
87pub struct IntervalYM {
88    pub(crate) intvl: dpiIntervalYM,
89    precision: u8,
90}
91
92impl IntervalYM {
93    fn check(years: i32, months: i32) -> Result<()> {
94        let errmsg = if !(-999999999..=999999999).contains(&years) {
95            "years must be between -999999999 and 999999999"
96        } else if !(-11..=11).contains(&months) {
97            "months must be between -11 and 11"
98        } else if (years >= 0 && months >= 0) || (years <= 0 && months <= 0) {
99            return Ok(());
100        } else {
101            "years and months must be zeor or positive; or zero or negative"
102        };
103        Err(Error::out_of_range(format!(
104            "{errmsg} but years={years}, months={months}"
105        )))
106    }
107
108    pub(crate) fn from_dpi_interval_ym(
109        it: &dpiIntervalYM,
110        oratype: &OracleType,
111    ) -> Result<IntervalYM> {
112        let prec = match *oratype {
113            OracleType::IntervalYM(prec) => prec,
114            _ => 2,
115        };
116        IntervalYM::new(it.years, it.months)?.and_prec(prec)
117    }
118
119    /// Creates a new IntervalYM.
120    ///
121    /// Valid values are:
122    ///
123    /// | argument | valid values |
124    /// |---|---|
125    /// | `years` | -999999999 to 999999999 |
126    /// | `months` | -11 to 11 |
127    ///
128    /// All arguments must be zero or positive to create a positive interval.
129    /// All arguments must be zero or negative to create a negative interval.
130    pub fn new(years: i32, months: i32) -> Result<IntervalYM> {
131        Self::check(years, months)?;
132        Ok(IntervalYM {
133            intvl: dpiIntervalYM { years, months },
134            precision: 9,
135        })
136    }
137
138    /// Creates a new IntervalYM with precision.
139    ///
140    /// The precision affects text representation of IntervalYM.
141    /// It doesn't affect comparison.
142    pub fn and_prec(&self, precision: u8) -> Result<IntervalYM> {
143        if precision > 9 {
144            Err(Error::out_of_range(format!(
145                "precision must be 0 to 9 but {}",
146                precision
147            )))
148        } else {
149            Ok(IntervalYM { precision, ..*self })
150        }
151    }
152
153    /// Returns years component.
154    pub fn years(&self) -> i32 {
155        self.intvl.years
156    }
157
158    /// Returns months component.
159    pub fn months(&self) -> i32 {
160        self.intvl.months
161    }
162
163    /// Returns precision.
164    pub fn precision(&self) -> u8 {
165        self.precision
166    }
167}
168
169impl cmp::PartialEq for IntervalYM {
170    fn eq(&self, other: &Self) -> bool {
171        self.intvl.years == other.intvl.years && self.intvl.months == other.intvl.months
172    }
173}
174
175impl fmt::Display for IntervalYM {
176    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
177        if self.intvl.years < 0 || self.intvl.months < 0 {
178            write!(f, "-")?;
179        } else {
180            write!(f, "+")?;
181        }
182        let years = self.intvl.years.abs();
183        match self.precision {
184            2 => write!(f, "{:02}", years)?,
185            3 => write!(f, "{:03}", years)?,
186            4 => write!(f, "{:04}", years)?,
187            5 => write!(f, "{:05}", years)?,
188            6 => write!(f, "{:06}", years)?,
189            7 => write!(f, "{:07}", years)?,
190            8 => write!(f, "{:08}", years)?,
191            9 => write!(f, "{:09}", years)?,
192            _ => write!(f, "{}", years)?,
193        };
194        write!(f, "-{:02}", self.intvl.months.abs())
195    }
196}
197
198impl str::FromStr for IntervalYM {
199    type Err = ParseOracleTypeError;
200
201    fn from_str(s: &str) -> result::Result<Self, Self::Err> {
202        let err = || ParseOracleTypeError::new("IntervalYM");
203        let mut s = Scanner::new(s);
204        let minus = match s.char() {
205            Some('+') => {
206                s.next();
207                false
208            }
209            Some('-') => {
210                s.next();
211                true
212            }
213            _ => false,
214        };
215        let years: i32 = s
216            .read_digits()
217            .ok_or_else(err)?
218            .try_into()
219            .map_err(|_| err())?;
220        let precision = s.ndigits().try_into().map_err(|_| err())?;
221        if let Some('-') = s.char() {
222            s.next();
223        } else {
224            return Err(err());
225        }
226        let months: i32 = s
227            .read_digits()
228            .ok_or_else(err)?
229            .try_into()
230            .map_err(|_| err())?;
231        if s.char().is_some() {
232            return Err(err());
233        }
234        IntervalYM::new(
235            if minus { -years } else { years },
236            if minus { -months } else { months },
237        )
238        .map_err(|_| err())?
239        .and_prec(precision)
240        .map_err(|_| err())
241    }
242}

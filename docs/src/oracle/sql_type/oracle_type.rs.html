28// NativeType corresponds to dpiNativeTypeNum in ODPI
29// except Char, Number, Raw, CLOB and BLOB.
30#[derive(Debug, Clone, PartialEq)]
31pub enum NativeType {
32    Int64,      // i64 in rust
33    UInt64,     // u64 in rust
34    Float,      // f32 in rust
35    Double,     // f64 in rust
36    Char,       // String or `str in rust
37    Number,     // string represention of Oracle number
38    Raw,        // Vec<u8> in rust
39    Timestamp,  // oracle::Timestamp in rust
40    IntervalDS, // oracle::IntervalDS in rust
41    IntervalYM, // oracle::IntervalYM in rust
42    Clob,
43    Blob,
44    Object(ObjectType),
45    Stmt,
46    Boolean, // bool in rust
47    Rowid,
48    Vector,
49}

51impl NativeType {
52    pub fn to_native_type_num(&self) -> dpiNativeTypeNum {
53        match *self {
54            NativeType::Int64 => DPI_NATIVE_TYPE_INT64,
55            NativeType::UInt64 => DPI_NATIVE_TYPE_UINT64,
56            NativeType::Float => DPI_NATIVE_TYPE_FLOAT,
57            NativeType::Double => DPI_NATIVE_TYPE_DOUBLE,
58            NativeType::Char => DPI_NATIVE_TYPE_BYTES,
59            NativeType::Number => DPI_NATIVE_TYPE_BYTES,
60            NativeType::Raw => DPI_NATIVE_TYPE_BYTES,
61            NativeType::Timestamp => DPI_NATIVE_TYPE_TIMESTAMP,
62            NativeType::IntervalDS => DPI_NATIVE_TYPE_INTERVAL_DS,
63            NativeType::IntervalYM => DPI_NATIVE_TYPE_INTERVAL_YM,
64            NativeType::Clob => DPI_NATIVE_TYPE_LOB,
65            NativeType::Blob => DPI_NATIVE_TYPE_LOB,
66            NativeType::Object(_) => DPI_NATIVE_TYPE_OBJECT,
67            NativeType::Stmt => DPI_NATIVE_TYPE_STMT,
68            NativeType::Boolean => DPI_NATIVE_TYPE_BOOLEAN,
69            NativeType::Rowid => DPI_NATIVE_TYPE_ROWID,
70            NativeType::Vector => DPI_NATIVE_TYPE_VECTOR,
71        }
72    }

74    pub(crate) fn to_object_type_handle(&self) -> *mut dpiObjectType {
75        match *self {
76            NativeType::Object(ref objtype) => objtype.handle().raw(),
77            _ => ptr::null_mut(),
78        }
79    }

82/// Raw data inside of [`SqlValue`]
83///
84/// When the data type of an enum variant field is a reference,
85/// the field refers to the internal buffer provided by ODPI-C.
86///
87/// It is intended to be used in the following case.
88///
89/// Assuming that you have a decimal class `Decimal` which has `from_str`
90/// and `from_i64` methods and try to get Oralce numbers as the class
91/// without precision loss, you need to get the SQL value as `String`
92/// and convert it by `from_str` without using this type.
93///
94/// For example
95/// ```no_run
96/// # use oracle::{Error, Result};
97/// # use oracle::test_util;
98/// # struct Decimal();
99/// # impl Decimal { fn from_str(_: &str) { unimplemented!() } }
100/// # let conn = test_util::connect()?;
101/// for number_col_result in conn.query_as::<String>("select number_column from table_name", &[])? {
102///    let number_col = number_col_result?;
103///    let decimal_value = Decimal::from_str(&number_col);
104/// }
105/// # Ok::<(), Error>(())
106/// ```
107/// Another example
108/// ```no_run
109/// # use oracle::{Error, Result, SqlValue};
110/// # use oracle::sql_type::FromSql;
111/// # use oracle::test_util;
112/// # struct Decimal();
113/// # impl Decimal { fn from_str(_: &str) -> Decimal { unimplemented!() } }
114/// impl FromSql for Decimal {
115///     fn from_sql(val: &SqlValue) -> Result<Decimal> {
116///         Ok(Decimal::from_str(&val.get::<String>()?))
117///     }
118/// }
119/// # let conn = test_util::connect()?;
120/// for decimal_value_result in conn.query_as::<Decimal>("select number_column from table_name", &[])? {
121///    let decimal_value = decimal_value_result?;
122/// }
123/// # Ok::<(), Error>(())
124/// ```
125/// These codes are inefficient because `String` values are created just to be passed to `from_str()`.
126///
127/// By using the `InnerValue` type, you can refer to the internal `str` value directly and avoid
128/// temporary memory allocation.
129/// ```no_run
130/// # use oracle::{Error, ErrorKind, Result, SqlValue};
131/// # use oracle::sql_type::{FromSql, InnerValue};
132/// # use oracle::test_util;
133/// # struct Decimal();
134/// # impl Decimal {
135/// #     fn from_str(_: &str) -> Result<Decimal> { unimplemented!() }
136/// #     fn from_i64(_: i64) -> Decimal { unimplemented!() }
137/// # }
138/// impl FromSql for Decimal {
139///     fn from_sql(val: &SqlValue) -> Result<Decimal> {
140///         match val.as_inner_value()? {
141///             InnerValue::Int64(val) => Ok(Decimal::from_i64(val)),
142///             InnerValue::Number(val) => Decimal::from_str(val).map_err(|err|
143///                Error::with_source(ErrorKind::InvalidTypeConversion, err)
144///             ),
145///             _ => Err(Error::new(
146///                 ErrorKind::InvalidTypeConversion,
147///                 format!("Cannot convert {} to Decimal", val.oracle_type()?)
148///             )),
149///         }
150///     }
151/// }
152/// # let conn = test_util::connect()?;
153/// for decimal_value_result in conn.query_as::<Decimal>("select number_column from table_name", &[])? {
154///    let decimal_value = decimal_value_result?;
155/// }
156/// # Ok::<(), Error>(())
157/// ```
158#[derive(Clone, Debug)]
159#[non_exhaustive]
160pub enum InnerValue<'a> {
161    Int64(i64),
162    UInt64(u64),
163    Float(f32),
164    Double(f64),
165    Char(&'a [u8]),
166    Number(&'a str),
167    Raw(&'a [u8]),
168    Timestamp(&'a dpiTimestamp),
169    IntervalDS(&'a dpiIntervalDS),
170    IntervalYM(&'a dpiIntervalYM),
171    Clob(*mut dpiLob),
172    Blob(*mut dpiLob),
173    Object(*mut dpiObject),
174    Stmt(*mut dpiStmt),
175    Boolean(bool),
176    Rowid(*mut dpiRowid),
177    Vector(*mut dpiVector),
178}

215/// Oracle data type
216#[derive(Debug, Clone, PartialEq)]
217#[non_exhaustive]
218pub enum OracleType {
219    /// VARCHAR2(size)
220    Varchar2(u32),
221
222    /// NVARCHAR2(size)
223    NVarchar2(u32),
224
225    /// CHAR(size)
226    Char(u32),
227
228    /// NCHAR(size)
229    NChar(u32),
230
231    /// ROWID
232    Rowid,
233
234    /// RAW(size)
235    Raw(u32),
236
237    /// BINARY_FLOAT
238    ///
239    /// IEEE 754 single-precision (32-bit) floating-point number
240    BinaryFloat,
241
242    /// BINARY_DOUBLE
243    ///
244    /// IEEE 754 double-precision (64-bit) floating-point number
245    BinaryDouble,
246
247    /// NUMBER(precision, scale)
248    ///
249    /// `precision` is between 0 and 38. When it is 0, its actual precision is
250    /// 38 and `(precision, scale)` is omitted in text represention.
251    ///
252    /// `scale` is between -87 and 127. When it is 0, this is represented
253    /// as `NUMBER(precision)` in text.
254    Number(u8, i8),
255
256    /// FLOAT(precision)
257    ///
258    /// This is a subtype of NUMBER. The internal format is same with NUMBER,
259    /// which means that numbers are stored as decimal not as binary.
260    /// Use BINARY_DOUBLE or BINARY_FLOAT to store f64 or f32 rust types.
261    ///
262    /// `precision` is between 0 and 126. When it is 126, `(precision)` is
263    /// omitted in text represention.
264    Float(u8),
265
266    /// DATE data type
267    Date,
268
269    /// TIMESTAMP(fsprec)
270    ///
271    /// Timestamp data type without time zone.
272    ///
273    /// `fsprec` is fractional seconds precision between 0 and 9. When it is
274    /// 6, `(fsprec)` is omitted in text represention.
275    Timestamp(u8),
276
277    /// TIMESTAMP(fsprec) WITH TIME ZONE
278    ///
279    /// Timestamp data type with time zone.
280    ///
281    /// `fsprec` is fractional seconds precision between 0 and 9. When it is
282    /// 6, `(fsprec)` is omitted in text represention.
283    TimestampTZ(u8),
284
285    /// TIMESTAMP(fsprec) WITH LOCAL TIME ZONE
286    ///
287    /// Timestamp data type in local session time zone. Clients in different
288    /// session time zones retrieves different timestamp.
289    ///
290    /// `fsprec` is fractional seconds precision between 0 and 9. When it is
291    /// 6, `(fsprec)` is omitted in text represention.
292    TimestampLTZ(u8),
293
294    /// INTERVAL DAY(lfprec) TO SECOND(fsprec)
295    ///
296    /// `lfprec` is leading field precision between 0 and 9. When it is 2,
297    /// `(lfprec)` is omitted in text represention.
298    ///
299    /// `fsprec` is fractional seconds precision between 0 and 9. When it is
300    /// 6, `(fsprec)` is omitted in text represention.
301    IntervalDS(u8, u8),
302
303    /// INTERVAL YEAR(lfprec) TO MONTH
304    ///
305    /// `lfprec` is leading field precision between 0 and 9. When it is 2,
306    /// `(lfprec)` is omitted in text represention.
307    IntervalYM(u8),
308
309    /// CLOB
310    CLOB,
311
312    /// NCLOB
313    NCLOB,
314
315    /// BLOB
316    BLOB,
317
318    /// BFILE
319    BFILE,
320
321    /// REF CURSOR (not supported)
322    RefCursor,
323
324    /// BOOLEAN (not supported)
325    Boolean,
326
327    /// Object
328    Object(ObjectType),
329
330    /// LONG
331    Long,
332
333    /// LONG RAW
334    LongRaw,
335
336    /// JSON data type introduced in Oracle 21c
337    Json,
338
339    /// XML
340    Xml,
341
342    /// [VECTOR] data type
343    ///
344    /// The first tuple element is number of dimensions. `0` corresponds to `*` in data type definition.
345    ///
346    /// The second is dimension element format. `VecFmt::Flexible` corresponds to `*`.
347    ///
348    /// **Examples:**
349    ///
350    /// Definition in SQL | value in the oracle crate
351    /// ---|---
352    /// `VECTOR`<br/>`VECTOR(*, *)` | `OracleType::Vector(0, VecFmt::Flexible)`
353    /// `VECTOR(*, FLOAT32)` | `OracleType::Vector(0, VecFmt::Float32)`
354    /// `VECTOR(4096)`<br/>`VECTOR(4096, *)` | `OracleType::Vector(4096, VecFmt::Flexible)`
355    /// `VECTOR(8192, FLOAT64)` | `OracleType::Vector(8192, VecFmt::Float64)`
356    ///
357    /// [VECTOR]: https://docs.oracle.com/en/database/oracle/oracle-database/23/vecse/overview-ai-vector-search.html
358    #[non_exhaustive]
359    Vector(u32, VecFmt),
360
361    /// Integer type in Oracle object type attributes. This will be renamed to Integer in future.
362    Int64,
363
364    /// Not an Oracle type, used only internally to bind/define values as u64
365    UInt64,
366}

368impl OracleType {
369    pub(crate) fn from_type_info(conn: &Conn, info: &dpiDataTypeInfo) -> Result<OracleType> {
370        match info.oracleTypeNum {
371            DPI_ORACLE_TYPE_VARCHAR => Ok(OracleType::Varchar2(info.dbSizeInBytes)),
372            DPI_ORACLE_TYPE_NVARCHAR => Ok(OracleType::NVarchar2(info.sizeInChars)),
373            DPI_ORACLE_TYPE_CHAR => Ok(OracleType::Char(info.dbSizeInBytes)),
374            DPI_ORACLE_TYPE_NCHAR => Ok(OracleType::NChar(info.sizeInChars)),
375            DPI_ORACLE_TYPE_ROWID => Ok(OracleType::Rowid),
376            DPI_ORACLE_TYPE_RAW => Ok(OracleType::Raw(info.dbSizeInBytes)),
377            DPI_ORACLE_TYPE_NATIVE_FLOAT => Ok(OracleType::BinaryFloat),
378            DPI_ORACLE_TYPE_NATIVE_DOUBLE => Ok(OracleType::BinaryDouble),
379            DPI_ORACLE_TYPE_NATIVE_INT => Ok(OracleType::Int64),
380            DPI_ORACLE_TYPE_NUMBER => {
381                if info.precision != 0 && info.scale == -127 {
382                    Ok(OracleType::Float(info.precision as u8))
383                } else {
384                    Ok(OracleType::Number(info.precision as u8, info.scale))
385                }
386            }
387            DPI_ORACLE_TYPE_DATE => Ok(OracleType::Date),
388            DPI_ORACLE_TYPE_TIMESTAMP => Ok(OracleType::Timestamp(info.fsPrecision)),
389            DPI_ORACLE_TYPE_TIMESTAMP_TZ => Ok(OracleType::TimestampTZ(info.fsPrecision)),
390            DPI_ORACLE_TYPE_TIMESTAMP_LTZ => Ok(OracleType::TimestampLTZ(info.fsPrecision)),
391            DPI_ORACLE_TYPE_INTERVAL_DS => Ok(OracleType::IntervalDS(
392                info.precision as u8,
393                info.fsPrecision,
394            )),
395            DPI_ORACLE_TYPE_INTERVAL_YM => Ok(OracleType::IntervalYM(info.precision as u8)),
396            DPI_ORACLE_TYPE_CLOB => Ok(OracleType::CLOB),
397            DPI_ORACLE_TYPE_NCLOB => Ok(OracleType::NCLOB),
398            DPI_ORACLE_TYPE_BLOB => Ok(OracleType::BLOB),
399            DPI_ORACLE_TYPE_BFILE => Ok(OracleType::BFILE),
400            DPI_ORACLE_TYPE_STMT => Ok(OracleType::RefCursor),
401            DPI_ORACLE_TYPE_BOOLEAN => Ok(OracleType::Boolean),
402            DPI_ORACLE_TYPE_OBJECT => Ok(OracleType::Object(ObjectType::from_dpi_object_type(
403                conn.clone(),
404                DpiObjectType::with_add_ref(info.objectType),
405            )?)),
406            DPI_ORACLE_TYPE_LONG_VARCHAR => Ok(OracleType::Long),
407            DPI_ORACLE_TYPE_LONG_RAW => Ok(OracleType::LongRaw),
408            DPI_ORACLE_TYPE_JSON => Ok(OracleType::Json),
409            DPI_ORACLE_TYPE_XMLTYPE => Ok(OracleType::Xml),
410            DPI_ORACLE_TYPE_VECTOR => Ok(OracleType::Vector(
411                info.vectorDimensions,
412                VecFmt::from_dpi(info.vectorFormat)?,
413            )),
414            _ => Err(Error::internal_error(format!(
415                "unknown Oracle type number {}",
416                info.oracleTypeNum
417            ))),
418        }
419    }

421    // Returns parameters to create a dpiVar handle.
422    pub(crate) fn var_create_param(&self) -> Result<VarParam> {
423        // The followings are basically same with dpiAllOracleTypes[] in
424        // dpiOracleType.c. If enum OracleType has an attribute corresponding
425        // to defaultNativeTypeNum of dpiQueryInfo, this mapping is not needed.
426        // However I don't want to do it to hide internal information such
427        // as dpiNativeTypeNum.
428        match *self {
429            OracleType::Varchar2(size) => {
430                Ok(VarParam::new(DPI_ORACLE_TYPE_VARCHAR, NativeType::Char).size(size))
431            }
432            OracleType::NVarchar2(size) => {
433                Ok(VarParam::new(DPI_ORACLE_TYPE_NVARCHAR, NativeType::Char).size(size))
434            }
435            OracleType::Char(size) => {
436                Ok(VarParam::new(DPI_ORACLE_TYPE_CHAR, NativeType::Char).size(size))
437            }
438            OracleType::NChar(size) => {
439                Ok(VarParam::new(DPI_ORACLE_TYPE_NCHAR, NativeType::Char).size(size))
440            }
441            OracleType::Rowid => Ok(VarParam::new(DPI_ORACLE_TYPE_ROWID, NativeType::Rowid)),
442            OracleType::Raw(size) => Ok(VarParam::new(DPI_ORACLE_TYPE_RAW, NativeType::Raw)
443                .size(size)
444                .size_is_byte()),
445            OracleType::BinaryFloat => Ok(VarParam::new(
446                DPI_ORACLE_TYPE_NATIVE_FLOAT,
447                NativeType::Float,
448            )),
449            OracleType::BinaryDouble => Ok(VarParam::new(
450                DPI_ORACLE_TYPE_NATIVE_DOUBLE,
451                NativeType::Double,
452            )),
453            OracleType::Number(_, _) | OracleType::Float(_) => {
454                Ok(VarParam::new(DPI_ORACLE_TYPE_NUMBER, NativeType::Number))
455            }
456            OracleType::Date => Ok(VarParam::new(DPI_ORACLE_TYPE_DATE, NativeType::Timestamp)),
457            OracleType::Timestamp(_) => Ok(VarParam::new(
458                DPI_ORACLE_TYPE_TIMESTAMP,
459                NativeType::Timestamp,
460            )),
461            OracleType::TimestampTZ(_) => Ok(VarParam::new(
462                DPI_ORACLE_TYPE_TIMESTAMP_TZ,
463                NativeType::Timestamp,
464            )),
465            OracleType::TimestampLTZ(_) => Ok(VarParam::new(
466                DPI_ORACLE_TYPE_TIMESTAMP_LTZ,
467                NativeType::Timestamp,
468            )),
469            OracleType::IntervalDS(_, _) => Ok(VarParam::new(
470                DPI_ORACLE_TYPE_INTERVAL_DS,
471                NativeType::IntervalDS,
472            )),
473            OracleType::IntervalYM(_) => Ok(VarParam::new(
474                DPI_ORACLE_TYPE_INTERVAL_YM,
475                NativeType::IntervalYM,
476            )),
477            OracleType::CLOB => Ok(VarParam::new(DPI_ORACLE_TYPE_CLOB, NativeType::Clob)),
478            OracleType::NCLOB => Ok(VarParam::new(DPI_ORACLE_TYPE_NCLOB, NativeType::Clob)),
479            OracleType::BLOB => Ok(VarParam::new(DPI_ORACLE_TYPE_BLOB, NativeType::Blob)),
480            OracleType::BFILE => Ok(VarParam::new(DPI_ORACLE_TYPE_BFILE, NativeType::Blob)),
481            OracleType::RefCursor => Ok(VarParam::new(DPI_ORACLE_TYPE_STMT, NativeType::Stmt)),
482            OracleType::Boolean => Ok(VarParam::new(DPI_ORACLE_TYPE_BOOLEAN, NativeType::Boolean)),
483            OracleType::Object(ref objtype) => Ok(VarParam::new(
484                DPI_ORACLE_TYPE_OBJECT,
485                NativeType::Object(objtype.clone()),
486            )),
487            OracleType::Long => Ok(VarParam::new(
488                DPI_ORACLE_TYPE_LONG_VARCHAR,
489                NativeType::Char,
490            )),
491            OracleType::LongRaw => Ok(VarParam::new(DPI_ORACLE_TYPE_LONG_RAW, NativeType::Raw)),
492            OracleType::Xml => Ok(VarParam::new(DPI_ORACLE_TYPE_XMLTYPE, NativeType::Char)),
493            OracleType::Vector(_, format) => {
494                Ok(VarParam::new(DPI_ORACLE_TYPE_VECTOR, NativeType::Vector).vector_format(format))
495            }
496            OracleType::Int64 => Ok(VarParam::new(DPI_ORACLE_TYPE_NATIVE_INT, NativeType::Int64)),
497            OracleType::UInt64 => Ok(VarParam::new(
498                DPI_ORACLE_TYPE_NATIVE_UINT,
499                NativeType::UInt64,
500            )),
501            _ => Err(Error::internal_error(format!(
502                "unsupported Oracle type {}",
503                self
504            ))),
505        }
506    }
507}

509impl fmt::Display for OracleType {
510    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
511        match *self {
512            OracleType::Varchar2(size) => write!(f, "VARCHAR2({})", size),
513            OracleType::NVarchar2(size) => write!(f, "NVARCHAR2({})", size),
514            OracleType::Char(size) => write!(f, "CHAR({})", size),
515            OracleType::NChar(size) => write!(f, "NCHAR({})", size),
516            OracleType::Rowid => write!(f, "ROWID"),
517            OracleType::Raw(size) => write!(f, "RAW({})", size),
518            OracleType::BinaryFloat => write!(f, "BINARY_FLOAT"),
519            OracleType::BinaryDouble => write!(f, "BINARY_DOUBLE"),
520            OracleType::Number(prec, scale) => {
521                if prec == 0 {
522                    write!(f, "NUMBER")
523                } else if scale == 0 {
524                    write!(f, "NUMBER({})", prec)
525                } else {
526                    write!(f, "NUMBER({},{})", prec, scale)
527                }
528            }
529            OracleType::Float(prec) => {
530                if prec == 126 {
531                    write!(f, "FLOAT")
532                } else {
533                    write!(f, "FLOAT({})", prec)
534                }
535            }
536            OracleType::Date => write!(f, "DATE"),
537            OracleType::Timestamp(fsprec) => {
538                if fsprec == 6 {
539                    write!(f, "TIMESTAMP")
540                } else {
541                    write!(f, "TIMESTAMP({})", fsprec)
542                }
543            }
544            OracleType::TimestampTZ(fsprec) => {
545                if fsprec == 6 {
546                    write!(f, "TIMESTAMP WITH TIME ZONE")
547                } else {
548                    write!(f, "TIMESTAMP({}) WITH TIME ZONE", fsprec)
549                }
550            }
551            OracleType::TimestampLTZ(fsprec) => {
552                if fsprec == 6 {
553                    write!(f, "TIMESTAMP WITH LOCAL TIME ZONE")
554                } else {
555                    write!(f, "TIMESTAMP({}) WITH LOCAL TIME ZONE", fsprec)
556                }
557            }
558            OracleType::IntervalDS(lfprec, fsprec) => {
559                if lfprec == 2 && fsprec == 6 {
560                    write!(f, "INTERVAL DAY TO SECOND")
561                } else {
562                    write!(f, "INTERVAL DAY({}) TO SECOND({})", lfprec, fsprec)
563                }
564            }
565            OracleType::IntervalYM(lfprec) => {
566                if lfprec == 2 {
567                    write!(f, "INTERVAL YEAR TO MONTH")
568                } else {
569                    write!(f, "INTERVAL YEAR({}) TO MONTH", lfprec)
570                }
571            }
572            OracleType::CLOB => write!(f, "CLOB"),
573            OracleType::NCLOB => write!(f, "NCLOB"),
574            OracleType::BLOB => write!(f, "BLOB"),
575            OracleType::BFILE => write!(f, "BFILE"),
576            OracleType::RefCursor => write!(f, "REF CURSOR"),
577            OracleType::Boolean => write!(f, "BOOLEAN"),
578            OracleType::Object(ref ty) => write!(f, "{}.{}", ty.schema(), ty.name()),
579            OracleType::Long => write!(f, "LONG"),
580            OracleType::LongRaw => write!(f, "LONG RAW"),
581            OracleType::Json => write!(f, "JSON"),
582            OracleType::Xml => write!(f, "XML"),
583            OracleType::Vector(ndim, format) => {
584                if ndim == 0 {
585                    write!(f, "VECTOR(*, {:#})", format)
586                } else {
587                    write!(f, "VECTOR({}, {:#})", ndim, format)
588                }
589            }
590            OracleType::Int64 => write!(f, "INT64 used internally"),
591            OracleType::UInt64 => write!(f, "UINT64 used internally"),
592        }
593    }

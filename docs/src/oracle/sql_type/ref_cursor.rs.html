37/// Result set output by or returned by a PL/SQL block or a stored procedure
38///
39/// This struct has four query methods, which are similar to [`Statement`]'s query methods
40/// excluding `params` arguments. The latter methods internally execute statements with
41/// specified `params` and then get query results. On the other hand, the former
42/// get query results executed in a PL/SQL block or a stored procedure. So there are no
43/// `params` arguments in this struct.
44///
45/// When settings about queries are set to [`StatementBuilder`], they are
46/// also applied to ref cursors output by a PL/SQL. It is worth to call
47/// [`StatementBuilder::fetch_array_size(1)`][`StatementBuilder::fetch_array_size]
48/// in order to reduce memory usage when a ref cursor contains at most one row.
49///
50/// # Examples
51///
52/// Ref cursor as an output parameter
53/// ```
54/// # use oracle::Error;
55/// # use oracle::sql_type::RefCursor;
56/// # use oracle::test_util;
57/// # let conn = test_util::connect()?;
58/// let sql = r#"
59/// begin
60///   open :cursor for select IntCol, StringCol from TestStrings order by IntCol;
61/// end;
62/// "#;
63/// let mut stmt = conn.statement(sql).build()?;
64/// stmt.execute(&[&None::<RefCursor>])?;
65///
66/// let mut cursor: RefCursor = stmt.bind_value(1)?;
67/// let mut n = 1;
68/// for row_result in cursor.query_as::<(i32, String)>()? {
69///     let (int_col, string_col) = row_result?;
70///     assert_eq!(int_col, n);
71///     assert_eq!(string_col, format!("String {}", n));
72///     n += 1;
73/// }
74/// # Ok::<(), Error>(())
75/// ```
76///
77/// Ref cursor returned by a PL/SQL block
78/// ```
79/// # use oracle::Error;
80/// # use oracle::sql_type::RefCursor;
81/// # use oracle::test_util::{self, check_version, VER12_1};
82/// # let conn = test_util::connect()?;
83/// # if !check_version(&conn, &VER12_1, &VER12_1)? {
84/// #     return Ok(()); // skip this test
85/// # }
86/// let sql = r#"
87/// declare
88///   cursor1 SYS_REFCURSOR;
89/// begin
90///   open cursor1 for select IntCol, StringCol from TestStrings order by IntCol;
91///   dbms_sql.return_result(cursor1);
92/// end;
93/// "#;
94/// let mut stmt = conn.statement(sql).build()?;
95/// stmt.execute(&[])?;
96///
97/// // Get the result set.
98/// let mut opt_cursor = stmt.implicit_result()?;
99/// assert!(opt_cursor.is_some());
100/// let mut cursor = opt_cursor.unwrap();
101/// let mut n = 1;
102/// for row_result in cursor.query_as::<(i32, String)>()? {
103///     let (int_col, string_col) = row_result?;
104///     assert_eq!(int_col, n);
105///     assert_eq!(string_col, format!("String {}", n));
106///     n += 1;
107/// }
108/// # Ok::<(), Error>(())
109/// ```
110///
111pub struct RefCursor {
112    stmt: Stmt,
113}
114
115impl RefCursor {
116    pub(crate) fn from_handle(
117        conn: Conn,
118        handle: DpiStmt,
119        query_params: QueryParams,
120    ) -> Result<RefCursor> {
121        chkerr!(
122            conn.ctxt(),
123            dpiStmt_setFetchArraySize(handle.raw, query_params.fetch_array_size)
124        );
125        let mut num_query_columns = 0;
126        chkerr!(
127            conn.ctxt(),
128            dpiStmt_getNumQueryColumns(handle.raw, &mut num_query_columns)
129        );
130        let mut stmt = Stmt::new(conn, handle, query_params, "".into());
131        stmt.init_row(num_query_columns as usize)?;
132        Ok(RefCursor { stmt })
133    }

135    /// Gets rows as an iterator of [`Row`]s.
136    ///
137    /// # Examples
138    ///
139    /// ```
140    /// # use oracle::Error;
141    /// # use oracle::sql_type::RefCursor;
142    /// # use oracle::test_util;
143    /// # let conn = test_util::connect()?;
144    /// let sql = r#"
145    /// begin
146    ///   open :cursor for select IntCol, StringCol from TestStrings order by IntCol;
147    /// end;
148    /// "#;
149    /// let mut stmt = conn.statement(sql).build()?;
150    /// stmt.execute(&[&None::<RefCursor>])?;
151    ///
152    /// let mut cursor: RefCursor = stmt.bind_value(1)?;
153    /// let mut n = 1;
154    /// for row_result in cursor.query()? {
155    ///     let row = row_result?;
156    ///     let int_col: i32 = row.get(0)?;
157    ///     let string_col: String = row.get(1)?;
158    ///     assert_eq!(int_col, n);
159    ///     assert_eq!(string_col, format!("String {}", n));
160    ///     n += 1;
161    /// }
162    /// # Ok::<(), Error>(())
163    /// ```
164    pub fn query(&mut self) -> Result<ResultSet<Row>> {
165        Ok(ResultSet::<Row>::new(&mut self.stmt))
166    }

168    /// Gets rows as an itertor of the specified type.
169    ///
170    /// # Examples
171    ///
172    /// ```
173    /// # use oracle::Error;
174    /// # use oracle::sql_type::RefCursor;
175    /// # use oracle::test_util;
176    /// # let conn = test_util::connect()?;
177    /// let sql = r#"
178    /// begin
179    ///   open :cursor for select IntCol, StringCol from TestStrings order by IntCol;
180    /// end;
181    /// "#;
182    /// let mut stmt = conn.statement(sql).build()?;
183    /// stmt.execute(&[&None::<RefCursor>])?;
184    ///
185    /// let mut cursor: RefCursor = stmt.bind_value(1)?;
186    /// let mut n = 1;
187    /// for row_result in cursor.query_as::<(i32, String)>()? {
188    ///     let (int_col, string_col) = row_result?;
189    ///     assert_eq!(int_col, n);
190    ///     assert_eq!(string_col, format!("String {}", n));
191    ///     n += 1;
192    /// }
193    /// # Ok::<(), Error>(())
194    /// ```
195    pub fn query_as<T>(&mut self) -> Result<ResultSet<T>>
196    where
197        T: RowValue,
198    {
199        Ok(ResultSet::<T>::new(&mut self.stmt))
200    }

202    /// Gets one row as [`Row`].
203    ///
204    /// # Examples
205    ///
206    /// ```
207    /// # use oracle::Error;
208    /// # use oracle::sql_type::RefCursor;
209    /// # use oracle::test_util;
210    /// # let conn = test_util::connect()?;
211    /// let sql = r#"
212    /// begin
213    ///   open :cursor for select StringCol from TestStrings where IntCol = :IntCol;
214    /// end;
215    /// "#;
216    /// let mut stmt = conn.statement(sql).fetch_array_size(1).build()?;
217    /// stmt.execute(&[&None::<RefCursor>, &1])?;
218    ///
219    /// let mut cursor: RefCursor = stmt.bind_value(1)?;
220    /// let string_col: String = cursor.query_row()?.get(0)?;
221    /// assert_eq!(string_col, "String 1");
222    /// # Ok::<(), Error>(())
223    /// ```
224    pub fn query_row(&mut self) -> Result<Row> {
225        self.query()?.next().unwrap_or(Err(Error::no_data_found()))
226    }

228    /// Gets one row as the specified type.
229    ///
230    /// # Examples
231    ///
232    /// ```
233    /// # use oracle::Error;
234    /// # use oracle::sql_type::RefCursor;
235    /// # use oracle::test_util;
236    /// # let conn = test_util::connect()?;
237    /// let sql = r#"
238    /// begin
239    ///   open :cursor for select StringCol from TestStrings where IntCol = :IntCol;
240    /// end;
241    /// "#;
242    /// let mut stmt = conn.statement(sql).fetch_array_size(1).build()?;
243    /// stmt.execute(&[&None::<RefCursor>, &1])?;
244    ///
245    /// let mut cursor: RefCursor = stmt.bind_value(1)?;
246    /// assert_eq!(cursor.query_row_as::<String>()?, "String 1");
247    /// # Ok::<(), Error>(())
248    /// ```
249    pub fn query_row_as<T>(&mut self) -> Result<T>
250    where
251        T: RowValue,
252    {
253        self.query_as()?
254            .next()
255            .unwrap_or(Err(Error::no_data_found()))
256    }
257}

289    fn params_from_ref_cursor(cursor: &RefCursor) -> (LobBindType, QueryParams) {
290        let sql_value = &cursor.stmt.row.as_ref().unwrap().column_values[0];
291        (sql_value.lob_bind_type, sql_value.query_params.clone())
292    }

294    #[test]
295    fn out_ref_cursor() -> Result<()> {
296        let conn = test_util::connect()?;
297        let sql = "begin pkg_TestOutCursors.TestOutCursor(:1, :2); end;";
298        let mut stmt = conn.statement(sql).build()?;
299        stmt.execute(&[&1, &None::<RefCursor>])?;
300        let mut ref_cursor: RefCursor = stmt.bind_value(2)?;
301        let row = ref_cursor.query_row_as::<(i32, String)>()?;
302        assert_eq!(row.0, 1);
303        assert_eq!(row.1, "String 1");
304
305        let params = params_from_ref_cursor(&ref_cursor);
306        assert_eq!(params.0, LobBindType::Bytes);
307        assert_eq!(params.1.fetch_array_size, DPI_DEFAULT_FETCH_ARRAY_SIZE);
308        assert_eq!(params.1.prefetch_rows, None);
309        assert_eq!(params.1.lob_bind_type, LobBindType::Bytes);
310        Ok(())
311    }

313    #[test]
314    fn out_ref_cursor_with_statement_builder_options() -> Result<()> {
315        let conn = test_util::connect()?;
316        let sql = "begin pkg_TestOutCursors.TestOutCursor(:1, :2); end;";
317        let mut stmt = conn
318            .statement(sql)
319            .fetch_array_size(2)
320            .prefetch_rows(3)
321            .lob_locator()
322            .build()?;
323        stmt.execute(&[&1, &None::<RefCursor>])?;
324        let mut ref_cursor: RefCursor = stmt.bind_value(2)?;
325        let row = ref_cursor.query_row_as::<(i32, String)>()?;
326        assert_eq!(row.0, 1);
327        assert_eq!(row.1, "String 1");
328
329        // Options specified by StatementBuilder propagates to ref cursors.
330        let params = params_from_ref_cursor(&ref_cursor);
331        assert_eq!(params.0, LobBindType::Locator);
332        assert_eq!(params.1.fetch_array_size, 2);
333        assert_eq!(params.1.prefetch_rows, Some(3));
334        assert_eq!(params.1.lob_bind_type, LobBindType::Locator);
335        Ok(())
336    }

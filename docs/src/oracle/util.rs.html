31pub struct Scanner<'a> {
32    chars: str::Chars<'a>,
33    char: Option<char>,
34    ndigits: u32,
35}
36
37impl Scanner<'_> {
38    pub fn new(s: &str) -> Scanner {
39        let mut chars = s.chars();
40        let char = chars.next();
41        Scanner {
42            chars,
43            char,
44            ndigits: 0,
45        }
46    }
47
48    pub fn next(&mut self) -> Option<char> {
49        self.char = self.chars.next();
50        self.char
51    }
52
53    pub fn char(&self) -> Option<char> {
54        self.char
55    }
56
57    pub fn read_digits(&mut self) -> Option<u64> {
58        let mut num = 0;
59        self.ndigits = 0;
60        loop {
61            num = num * 10
62                + match self.char {
63                    Some('0') => 0,
64                    Some('1') => 1,
65                    Some('2') => 2,
66                    Some('3') => 3,
67                    Some('4') => 4,
68                    Some('5') => 5,
69                    Some('6') => 6,
70                    Some('7') => 7,
71                    Some('8') => 8,
72                    Some('9') => 9,
73                    _ => {
74                        if self.ndigits > 0 {
75                            return Some(num);
76                        } else {
77                            return None;
78                        }
79                    }
80                };
81            self.char = self.chars.next();
82            self.ndigits += 1;
83        }
84    }
85
86    pub fn ndigits(&self) -> u32 {
87        self.ndigits
88    }
89}

91pub fn check_number_format(s: &str) -> result::Result<(), ParseOracleTypeError> {
92    let err = || ParseOracleTypeError::new("Oracle number");
93    let mut s = Scanner::new(s);
94
95    // optional negative sign
96    if let Some('-') = s.char() {
97        s.next();
98    }
99
100    // decimal part
101    if s.read_digits().is_none() {
102        return Err(err());
103    }
104    // optional fractional part
105    if let Some('.') = s.char() {
106        s.next();
107        if s.read_digits().is_none() {
108            return Err(err());
109        }
110    }
111    // an optional exponent
112    match s.char() {
113        Some('e') | Some('E') => {
114            s.next();
115            match s.char() {
116                Some('+') | Some('-') => {
117                    s.next();
118                }
119                _ => (),
120            }
121            if s.read_digits().is_none() {
122                return Err(err());
123            }
124        }
125        _ => (),
126    }
127    if s.char().is_some() {
128        return Err(err());
129    }
130    Ok(())
131}

133pub fn parse_str_into_raw(s: &str) -> result::Result<Vec<u8>, ParseOracleTypeError> {
134    let mut vec: Vec<u8> = Vec::with_capacity((s.len() + 1) / 2);
135    let mut upper = s.len() % 2 == 0; // set upper half
136    let mut upper_half = 0u8;
137    for chr in s.bytes() {
138        let half_byte = match chr {
139            b'0'..=b'9' => chr - b'0',
140            b'A'..=b'F' => chr - b'A' + 10,
141            b'a'..=b'f' => chr - b'a' + 10,
142            _ => return Err(ParseOracleTypeError::new("raw")),
143        };
144        if upper {
145            upper_half = half_byte << 4;
146        } else {
147            vec.push(upper_half + half_byte);
148        }
149        upper = !upper;
150    }
151    Ok(vec)
152}

154pub fn set_hex_string(s: &mut String, bytes: &[u8]) {
155    let to_hex = |x| {
156        if x < 10 {
157            (b'0' + x) as char
158        } else {
159            (b'A' + (x - 10)) as char
160        }
161    };
162    for byte in bytes {
163        s.push(to_hex(byte >> 4));
164        s.push(to_hex(byte & 0xF));
165    }
166}

168pub fn write_literal(
169    f: &mut fmt::Formatter,
170    s: &Result<String>,
171    oratype: &OracleType,
172) -> fmt::Result {
173    match s {
174        Ok(s) => match *oratype {
175            OracleType::Varchar2(_)
176            | OracleType::NVarchar2(_)
177            | OracleType::Char(_)
178            | OracleType::NChar(_)
179            | OracleType::Rowid
180            | OracleType::Raw(_)
181            | OracleType::CLOB
182            | OracleType::NCLOB
183            | OracleType::BLOB
184            | OracleType::BFILE
185            | OracleType::Long
186            | OracleType::LongRaw => {
187                write!(f, "\"")?;
188                for c in s.chars() {
189                    if c == '"' {
190                        write!(f, "\"")?;
191                    }
192                    write!(f, "{}", c)?;
193                }
194                write!(f, "\"")
195            }
196            _ => write!(f, "{}", s),
197        },
198        Err(err) if err.kind() == ErrorKind::NullValue => write!(f, "NULL"),
199        Err(err) => write!(f, "ERR({})", err),
200    }
201}

203pub fn string_into_c_string(s: String, name: &str) -> Result<CString> {
204    CString::new(s).map_err(|err| {
205        Error::invalid_argument(format!("{} cannot contain nul characters", name)).add_source(err)
206    })
207}

209#[cfg(test)]
210mod tests {

213    #[test]
214    fn test_scanner() {
215        let mut s = Scanner::new("123.4567890");
216        assert_eq!(s.read_digits(), Some(123));
217        assert_eq!(s.read_digits(), None);
218        assert_eq!(s.char(), Some('.'));
219        s.next();
220        assert_eq!(s.read_digits(), Some(4567890));
221        assert_eq!(s.char(), None);
222    }
223
224    #[test]
225    fn test_check_number_format() {
226        let ok = Ok(());
227        let err = Err(ParseOracleTypeError::new("Oracle number"));
228        assert_eq!(check_number_format("123"), ok);
229        assert_eq!(check_number_format("-123"), ok);
230        assert_eq!(check_number_format("-123."), err);
231        assert_eq!(check_number_format("-123.5"), ok);
232        assert_eq!(check_number_format("-123e"), err);
233        assert_eq!(check_number_format("-123e1"), ok);
234        assert_eq!(check_number_format("-123e+1"), ok);
235        assert_eq!(check_number_format("-123e-1"), ok);
236        assert_eq!(check_number_format("-123e-10"), ok);
237        assert_eq!(check_number_format(".123"), err);
238        assert_eq!(check_number_format("0.123"), ok);
239        assert_eq!(check_number_format(" 123"), err);
240        assert_eq!(check_number_format(""), err);
241        assert_eq!(check_number_format("a"), err);
242        assert_eq!(check_number_format("0.0"), ok);
243        assert_eq!(check_number_format("9.9"), ok);
244    }

246    #[test]
247    fn test_parse_str_into_raw() {
248        let err = Err(ParseOracleTypeError::new("raw"));
249        assert_eq!(parse_str_into_raw(""), Ok(vec![]));
250        assert_eq!(parse_str_into_raw("010203"), Ok(vec![1, 2, 3]));
251        assert_eq!(parse_str_into_raw("10203"), Ok(vec![1, 2, 3]));
252        assert_eq!(
253            parse_str_into_raw("090a0A0f0F"),
254            Ok(vec![9, 10, 10, 15, 15])
255        );
256        assert_eq!(parse_str_into_raw("G"), err);
257        assert_eq!(parse_str_into_raw("g"), err);
258        assert_eq!(
259            parse_str_into_raw("1223344556677889"),
260            Ok(vec![0x12, 0x23, 0x34, 0x45, 0x56, 0x67, 0x78, 0x89])
261        );
262        assert_eq!(
263            parse_str_into_raw("9aabbccddeeff0"),
264            Ok(vec![0x9a, 0xab, 0xbc, 0xcd, 0xde, 0xef, 0xf0])
265        );
266        assert_eq!(
267            parse_str_into_raw("9AABBCCDDEEFF0"),
268            Ok(vec![0x9a, 0xab, 0xbc, 0xcd, 0xde, 0xef, 0xf0])
269        );
270    }
